
IDIOMS YOU MUST USE INSTEAD OF String.get:
- Convert to chars: `let cs := s.data : List Char` and traverse/index the list; rebuild with `String.mk cs`.
- Or use slices: `s.take k` / `s.drop k` (characters counted in `Nat`).
Examples:
-- remove char at index i:
  let rec dropAt (xs : List Char) (i k : Nat) : List Char :=
    match xs with | [] => [] | c::xs =>
      let tail := dropAt xs i (k+1)
      if k = i then tail else c :: tail
  String.mk (dropAt s.data i 0)
-- rotate left by k:
  (s.drop k) ++ (s.take k)
